```javascript
    let node = this.head;
    let previous = null;

    while(node) {
        let next = node.next;
        node.next = previous;
        previous = node;
        node = next;
```

```javascript
/*

* 继续向后移动，b同理,当a或者b为null时，即a链表或b链表已到结尾，把node.next指向剩下有值的链表即可。
*
*/

    this.element = element;
    this.next = null;
};

function LList () {
    this.head = new Node('head');
    this.insert = insert;
}

function insert(newElement,current){
	var newNode = new Node(newElement);
	newNode.next = current.next;
	current.next = newNode;
    return newNode;

function join(list1,list2) {

    if(!list1 || !list2) {
        return;
    }

    let newList = new LList();
    let current = newList.head;

    let a = list1.head.next;
    let b = list2.head.next;

    while(b && a) {
        if( a.element < b.element ) {
            current.next = a;
            current = a;
            a = a.next;
        }else {
            current.next = b;
            current = b;
            b = b.next;
        };
    };

    if(!a) {
        current.next = b;
    };

    if(!b) {
        current.next = a;
    };

    return newList;
};
```

```javascript
    let current = list.head;
```

7. 输入两个链表，找出它们的第一个公共结点。
```javascript
/*
* 思路1：使用快慢指针，找到a链表和b链表的长度差diff，然后将指针移动到a和b相同长度的结点上，再开始一起移* 动找到相同结点
*
*/

	this.element = element;

        newNode = newElement;
    }
		newNode.next = current.next;
		current.next = newNode;

function findCommon(list1, list2) {
	if(!list1 || !list2) {
		return;
	};
	let listLen = 0,
			list2Len = 0;

	let node = list1.head;

	while(node) {
		++listLen;
		node = node.next;
	};

	node = list2.head;

	while(node) {
		++list2Len;
		node = node.next;
	};

	let rawDiff = listLen - list2Len;
	let a = list1.head,
			b = list2.head;
	let diff = Math.abs(rawDiff);

	if(rawDiff > 0) {
		while(diff) {
			--diff;
			a = a.next;
		};
	}

	if(rawDiff < 0) {
		while(diff) {
			--diff;
			b = b.next;
		};
	};

	while(a && b ) {
		if( a === b ) {
			return a;
		};
		a = a.next;
		b = b.next;
	};

	return null;
};
```
